********** Mapped Logic **********
FTCPE_cnt0: FTCPE port map (cnt(0),cnt_T(0),phase_in,reset,'0','1');
     cnt_T(0) <= NOT ((cnt(4) AND NOT cnt(0) AND cnt(1) AND cnt(2) AND cnt(3)));
FTCPE_cnt1: FTCPE port map (cnt(1),cnt_T(1),phase_in,reset,'0','1');
     cnt_T(1) <= ((cnt(0))
      OR (cnt(4) AND cnt(1) AND cnt(2) AND cnt(3)));
FTCPE_cnt2: FTCPE port map (cnt(2),cnt_T(2),phase_in,reset,'0','1');
     cnt_T(2) <= ((cnt(0) AND cnt(1))
      OR (cnt(4) AND cnt(1) AND cnt(2) AND cnt(3)));
FTCPE_cnt3: FTCPE port map (cnt(3),cnt_T(3),phase_in,reset,'0','1');
     cnt_T(3) <= ((cnt(0) AND cnt(1) AND cnt(2))
      OR (cnt(4) AND cnt(1) AND cnt(2) AND cnt(3)));
FTCPE_cnt4: FTCPE port map (cnt(4),cnt_T(4),phase_in,reset,'0','1');
     cnt_T(4) <= ((cnt(4) AND cnt(1) AND cnt(2) AND cnt(3))
      OR (cnt(0) AND cnt(1) AND cnt(2) AND cnt(3)));
FDCPE_div_out: FDCPE port map (div_out,NOT cnt(4),phase_in,'0','0',NOT reset);
